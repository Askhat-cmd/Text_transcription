## 1. Две важные особенности — прочитать перед реализацией

```text
ОСОБЕННОСТЬ 1 — почему здесь можно без сложной механики APPLY-22.
Правило R04 самодостаточно: если answer_obligation не равен
"provide_one_bounded_practice", ничего не происходит — ни один
last_debug-ключ не пишется, поток идёт дальше к R07 (694). Если
равен — есть РОВНО ОДНА последующая запись в last_debug на каждую
из трёх исходов, и она не конкурирует ни с чем извне (ни до, ни
после в пределах этого правила). Поэтому здесь безопасно сделать
helper ЧИСТОЙ функцией-классификатором без единого self-вызова —
это проще и безопаснее, чем тащить механику "патч + флаг" из
APPLY-22 туда, где в ней нет необходимости. Если в будущих срезах
семьи 2 (R07-R16) появится похожая конкуренция last_debug между
соседними правилами — применять полную механику (d) из APPLY-22,
а не эту упрощённую.

ОСОБЕННОСТЬ 2 — точная классификация трёх исходов.
Условие внешнего if (строки 673-679: "?" in text OR len(text.strip())
> 900 OR practice_multistep OR not practice_step_present OR not
practice_anchor_present) — назовём need_repair. Если need_repair
истинно И "будь сильным" in lowered_user -> исход "be_strong". Если
need_repair истинно И НЕ "будь сильным" -> исход "defer_repair". Если
need_repair ложно -> исход "strip_followup". Если answer_obligation
не совпал вообще -> исход "not_matched". Ровно 4 значения, ничего
среднего.
```

from __future__ import annotations

import re
from dataclasses import dataclass
from typing import Literal

from .writer_agent_constants import _contains_any


BoundedPracticeOutcome = Literal[
    "not_matched", "be_strong", "defer_repair", "strip_followup"
]


@dataclass(frozen=True)
class EnforceSlice3BoundedPracticeResult:
    outcome: BoundedPracticeOutcome


def _classify_enforce_slice3_bounded_practice(
    *,
    text: str,
    lowered_user: str,
    lowered_text: str,
    answer_obligation: str,
) -> EnforceSlice3BoundedPracticeResult:
    if answer_obligation != "provide_one_bounded_practice":
        return EnforceSlice3BoundedPracticeResult(outcome="not_matched")
    practice_anchor_present = _contains_any(
        lowered_text,
        ("будь сильным", "драйвер", "сильн", "напряж", "сдерж"),
    )
    practice_step_present = _contains_any(
        lowered_text,
        ("сделай", "заметь", "отметь", "поймай", "назови", "остановись", "выдох"),
    )
    practice_multistep = len(re.findall(r"(?m)^\s*(?:[-*•]|\d+[.)])\s+", text)) > 1
    if (
        "?" in text
        or len(text.strip()) > 900
        or practice_multistep
        or not practice_step_present
        or not practice_anchor_present
    ):
        if "будь сильным" in lowered_user:
            return EnforceSlice3BoundedPracticeResult(outcome="be_strong")
        return EnforceSlice3BoundedPracticeResult(outcome="defer_repair")
    return EnforceSlice3BoundedPracticeResult(outcome="strip_followup")
